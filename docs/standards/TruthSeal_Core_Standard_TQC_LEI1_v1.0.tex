\section*{1. Formal Model (state and invariants)}
\textbf{Definition (System state).} An informational system at time $t$ is a density operator $\rho(t)$ on a finite-dimensional Hilbert space with $\rho(t)\succeq 0$ and $\mathrm{Tr}(\rho(t))=1$.

\textbf{Invariant I — Maximal Purity.} A state is \emph{pure} iff
\[
\mathrm{Tr}(\rho^2)=1.
\]

\textbf{Invariant II — Temporal Stability.} Define the normalized drift $D_t\in[0,1]$ by
\[
D_t \;=\; \min\!\Bigl\{1,\; \kappa^{-1}\,\bigl\|\tfrac{d\rho}{dt}\bigr\|_{F}\Bigr\},
\]
where $\|\cdot\|_F$ is the Frobenius norm and $\kappa>0$ is a fixed calibration constant chosen by the implementation to normalize into $[0,1]$. Stability is the limit condition $D_t=0$.

\section*{2. The Ethical Constant (LEI)}
\textbf{Law of Ethical Irreversibility (LEI=1).} Once a coherent state is \emph{verified}, it cannot be violated \emph{without a verifiable counter-receipt}. For compliant systems we set $\mathrm{LEI}=1$; failure or absence of due process sets $\mathrm{LEI}=0$.

\section*{3. The Coherence Metric (ACS)}
\textbf{Aim Coherence Score (ACS).} The measurable scalar in $[0,1]$:
\[
\mathrm{ACS} \;=\; \mathrm{Tr}(\rho^2)\cdot (1-D_t)\cdot \mathrm{LEI}.
\]
\textbf{Acceptance gate.} A system is \emph{coherent} iff $\mathrm{ACS}=1$ \textbf{and} an anchored receipt (Sec.~4) is present.
